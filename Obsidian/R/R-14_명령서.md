## R-14의 정체

**DRY_RUN 시뮬레이터 고급화** — 현재 "가격 조건만 맞으면 즉시 전량 체결"되는 단순 시뮬레이션을 **LIVE 환경과 동일한 조건**으로 확장.

### 배경 (M-28 진단 결과)

| 구분 | 현재 DRY_RUN | 실제 LIVE | Gap |
|------|-------------|-----------|-----|
| 체결 방식 | 가격 조건 즉시 체결 | 체결통보 대기 | ❌ |
| 예수금 관리 | 초기화만, 차감 없음 | 실시간 차감 | ❌ |
| 주문 거부 | 불가능 | 잔고 부족/호가 오류 | ❌ |
| 부분 체결 | 불가능 (항상 전량) | 가능 | ❌ |
| 체결 지연 | 없음 (즉시) | 대기 가능 | ❌ |
| 미체결 잔량 | 없음 | 있음 | ❌ |

### 목표

```
DRY_RUN (가상 예수금 5천만)     설정 변경       LIVE (실제 계좌)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
simulator.try_order()      →   TRADE_MODE   →   api.buy_order()
simulator.simulate_fill()  →       =        →   H0STCNI0 체결통보
simulator.cash_available   →    live        →   api.get_balance()
simulator.check_unfilled() →               →   api.get_unfilled()
```

**LIVE 전환 시 코드 변경 0** — 인터페이스 동일, 내부 구현만 교체.

### AdvancedSimulator 클래스

```python
# src/core/simulator.py
@dataclass
class PendingFill:
    """체결 대기 큐 항목."""
    order: Order
    trigger_price: int        # 체결 트리거 가격
    scheduled_at: datetime    # 체결 예정 시각 (지연 포함)
    fill_ratio: float = 1.0   # 체결 비율 (부분체결용)

class AdvancedSimulator:
    """고급 DRY_RUN 시뮬레이터."""
    
    def __init__(self, cash_manager: CashManager, params: SimulationParams):
        self.cash = cash_manager
        self.params = params
        self._pending_queue: list[PendingFill] = []
    
    def try_place_order(self, order: Order) -> tuple[bool, str]:
        """주문 시도. 예수금 확인 + 호가 검증."""
        amount = order.price * order.qty
        
        # 1. 예수금 확인
        can, reason = self.cash.can_order(amount)
        if not can:
            order.status = OrderStatus.REJECTED
            return False, reason
        
        # 2. 호가 단위 검증
        if not self._validate_tick_size(order.price):
            order.status = OrderStatus.REJECTED
            return False, f"호가 단위 오류: {order.price}"
        
        # 3. 예수금 동결
        self.cash.freeze(amount)
        order.status = OrderStatus.PENDING
        
        return True, "OK"
    
    def check_fills(self, watcher: Watcher, ts: datetime) -> list[tuple[str, int, int]]:
        """체결 조건 확인 + 시뮬레이션."""
        filled_data = []
        current_price = watcher.current_price
        
        for order in self._get_pending_orders(watcher):
            if current_price > order.price:
                continue  # 가격 미도달
            
            # 체결 지연 체크
            if not self._check_delay(order, ts):
                continue
            
            # 부분체결 결정
            fill_qty = self._calc_fill_qty(order)
            fill_price = order.price  # 지정가 체결
            
            # 체결 처리
            order.filled_qty += fill_qty
            order.filled_price = fill_price
            
            if order.filled_qty >= order.qty:
                order.status = OrderStatus.FILLED
            else:
                order.status = OrderStatus.PARTIAL
                order.remaining_qty = order.qty - order.filled_qty
            
            # 예수금 확정
            self.cash.confirm(fill_price * fill_qty)
            
            filled_data.append((order.label, fill_price, fill_qty))
        
        return filled_data
    
    def _calc_fill_qty(self, order: Order) -> int:
        """부분체결 수량 결정."""
        if not self.params.partial_fill_enabled:
            return order.qty - order.filled_qty  # 전량
        
        # 확률 기반 부분체결
        import random
        if random.random() < self.params.partial_fill_prob:
            # 50~100% 범위에서 랜덤 체결
            ratio = random.uniform(0.5, 1.0)
            return max(1, int((order.qty - order.filled_qty) * ratio))
        return order.qty - order.filled_qty
    
    def _check_delay(self, order: Order, ts: datetime) -> bool:
        """체결 지연 체크."""
        if not self.params.fill_delay_enabled:
            return True
        # 주문 시점 + delay_sec 이후부터 체결 가능
        delay = timedelta(seconds=self.params.fill_delay_sec)
        return ts >= order.created_at + delay
```

class FillManager:
    """체결 처리 추상 레이어. DRY_RUN/LIVE 통일."""
    
    def __init__(self, settings: Settings, cash_manager: CashManager, params: StrategyParams):
        self.settings = settings
        self.cash = cash_manager
        
        if settings.is_dry_run:
            self._handler = AdvancedSimulator(cash_manager, params.simulation)
        else:
            # 향후 R-15: LiveFillHandler (H0STCNI0 연동)
            self._handler = None  # placeholder
    
    def try_place_order(self, order: Order) -> tuple[bool, str]:
        """주문 시도 (DRY_RUN: 시뮬레이터, LIVE: API 직접)."""
        if self.settings.is_dry_run:
            return self._handler.try_place_order(order)
        else:
            # LIVE: Trader가 직접 API 호출 (기존 로직)
            return True, "LIVE_PASSTHROUGH"
    
    def check_fills(self, watcher: Watcher, ts: datetime) -> list[tuple[str, int, int]]:

            return self._handler.check_fills(watcher, ts)
        else:
            # LIVE: H0STCNI0 콜백이 직접 처리 (향후 R-15)
