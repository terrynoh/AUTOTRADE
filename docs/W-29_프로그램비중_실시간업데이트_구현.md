## 작업 2: _recalc_prices() 메서드 분리

```python
# watcher.py — Watcher 클래스에 새 메서드 추가

def _recalc_prices(self) -> None:
    """is_double_digit 기준으로 매수가/손절가 재계산.
    
    R-13: 비중 변경 시 호출하여 가격 갱신.
    confirmed_high가 설정된 상태(TRIGGERED 이후)에서만 유효.
    """
    if self.confirmed_high <= 0:
        return  # TRIGGERED 이전이면 무시
    
    if self.market == MarketType.KOSPI:
        if self.is_double_digit:
            buy1_pct = self.params.entry.kospi_double_buy1_pct
            buy2_pct = self.params.entry.kospi_double_buy2_pct
            stop_pct = self.params.exit.kospi_double_hard_stop_pct
        else:
            buy1_pct = self.params.entry.kospi_single_buy1_pct
            buy2_pct = self.params.entry.kospi_single_buy2_pct
            stop_pct = self.params.exit.kospi_single_hard_stop_pct
    else:
        if self.is_double_digit:
            buy1_pct = self.params.entry.kosdaq_double_buy1_pct
            buy2_pct = self.params.entry.kosdaq_double_buy2_pct
            stop_pct = self.params.exit.kosdaq_double_hard_stop_pct
        else:
            buy1_pct = self.params.entry.kosdaq_single_buy1_pct
            buy2_pct = self.params.entry.kosdaq_single_buy2_pct
            stop_pct = self.params.exit.kosdaq_single_hard_stop_pct

    self.target_buy1_price = floor_to_tick(int(self.confirmed_high * (1 - buy1_pct / 100)))
    self.target_buy2_price = floor_to_tick(int(self.confirmed_high * (1 - buy2_pct / 100)))
    self.hard_stop_price_value = ceil_to_tick(int(self.confirmed_high * (1 - stop_pct / 100)))
```

### 6.2 구현

```python
# main.py — run() 함수 내부에 태스크 추가

async def _ratio_updater():
    """R-13: 프로그램 순매수 비중 실시간 업데이트.
    
    1초 간격으로 WATCHING/TRIGGERED/BUY1_FILLED 상태 Watcher의 비중 조회.
    비중 변경 시 is_double_digit 갱신 및 가격 재계산.
    """
    DOUBLE_THRESHOLD = 10.0  # 10% 이상이면 Double
    
    # 모의투자: 2.5초 간격 (rate limit 2건/초)
    interval = 2.5 if settings.is_paper else 1.0
    
    while True:
        try:
            await asyncio.sleep(interval)
            
            # 비중 업데이트 대상: WATCHING, TRIGGERED, BUY1_FILLED
            target_states = {
                WatcherState.WATCHING,
                WatcherState.TRIGGERED,
                WatcherState.BUY1_FILLED,
            }
            
            for w in coordinator.watchers:
                if w.state not in target_states:
                    continue
                
                try:
                    # 1. REST 호출
                    prog = await api.get_program_trade(w.code)
                    
                    # 2. 비중 계산
                    # get_current_price에서 trading_value 필요
                    price_info = await api.get_current_price(w.code)
                    trading_value = price_info.get("trading_value", 0)
                    
                    if trading_value > 0:
                        net_buy = prog.get("program_net_buy", 0)
                        new_ratio = (net_buy / trading_value) * 100
                    else:
                        new_ratio = 0.0
                    
                    # 3. 변경 감지
                    old_ratio = w.program_ratio
                    old_is_double = w.is_double_digit
                    new_is_double = new_ratio >= DOUBLE_THRESHOLD
                    
                    # 비중 갱신
                    w.program_ratio = new_ratio
                    
                    # 4. Double/Single 상태 변경 시 처리
                    if old_is_double != new_is_double:
                        w.is_double_digit = new_is_double
                        
                        logger.info(
                            f"[{w.name}] 비중 변경: {old_ratio:.1f}% → {new_ratio:.1f}% "
                            f"({'Double' if new_is_double else 'Single'})"
                        )
                        
                        # 5. 상태별 처리
                        if w.state == WatcherState.TRIGGERED:
                            # 1차 미체결 상태: 가격만 재계산
                            old_buy1 = w.target_buy1_price
                            old_stop = w.hard_stop_price_value
                            
                            w._recalc_prices()
                            
                            logger.info(
                                f"[{w.name}] 가격 재계산: "
                                f"buy1 {old_buy1:,} → {w.target_buy1_price:,}, "
                                f"stop {old_stop:,} → {w.hard_stop_price_value:,}"
                            )
                        
                        elif w.state == WatcherState.BUY1_FILLED:
                            # 1차 체결 상태: 2차 재발주 (A안 확정)
                            old_buy2 = w.target_buy2_price
                            old_stop = w.hard_stop_price_value
                            
                            w._recalc_prices()
                            
                            logger.info(
                                f"[{w.name}] 가격 재계산: "
                                f"buy2 {old_buy2:,} → {w.target_buy2_price:,}, "
                                f"stop {old_stop:,} → {w.hard_stop_price_value:,}"
                            )
                            
                            # 2차 미체결 시 재발주
                            if not w.buy2_filled and w.buy2_order_id:
                                balance = await api.get_balance()
                                cash = balance.get("available_cash", 0)
                                
                                success = await trader.cancel_and_reorder_buy2(
                                    w, w.target_buy2_price, cash
                                )
                                if success:
                                    logger.info(f"[{w.name}] buy2 재발주 완료")
                
                except Exception as e:
                    logger.warning(f"[{w.name}] 비중 업데이트 실패: {e}")
                    continue
        
        except asyncio.CancelledError:
            logger.info("_ratio_updater 종료")
            break
        except Exception as e:
            logger.error(f"_ratio_updater 에러: {e}")
            await asyncio.sleep(5)  # 에러 시 5초 대기 후 재시도
```

## 상태 전이 영향

### 비중 변경 시 영향받는 상태

| 상태 | 처리 |
|------|------|
| WATCHING | program_ratio만 갱신 (가격 계산 전이므로 무시) |
| TRIGGERED | _recalc_prices() 호출 (buy1/buy2/stop 재계산) |
| BUY1_FILLED | _recalc_prices() + cancel_and_reorder_buy2() |
| BUY2_FILLED | 무시 (모든 매수 완료) |
| LIQUIDATED | 무시 (청산 완료) |

### 손절선 완화 케이스 (수석님 확정)

> Double → Single 변경 시 손절선이 완화되면, 현재가가 기존 손절선 아래라도 **홀딩**.

이는 `_recalc_prices()` 호출 후 자연스럽게 적용됨:
- 기존: hard_stop_price_value = 72,000원
- 변경 후: hard_stop_price_value = 70,000원
- 현재가 71,500원이라도 새 손절선 아래가 아니므로 홀딩

## 검증

### 단위 테스트 시나리오

1. **TRIGGERED 상태에서 Double → Single**
   - buy1/buy2/stop 가격이 Single 기준으로 재계산되는지 확인

2. **BUY1_FILLED 상태에서 Single → Double**
   - buy2/stop 가격이 Double 기준으로 재계산되는지 확인

   - is_double_digit 유지, _recalc_prices() 미호출 확인
